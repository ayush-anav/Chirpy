# Dont forget, we require the POSTGRES DRIVER everytime for GO
### _ tells GO we will use the import as a side-effect (sql.Open())
```go
import(
_ "github.com/lib/pq" // postgres driver, _ means we will use as side-effect
)
```
```go
// loads our .env file so we can do os.Getenv
godotenv.Load()
dbURL := os.Getenv("DB_URL")

// uses the lib/pq driver to prepare to connect to db (connects lazily when first query runs)
db, dbErr := sql.Open("postgres", dbURL)

if dbErr != nil {
    log.Fatalf("could not open DB connection at URL: %s", dbErr)
}

// wires our sql queries to our db connection
dbQueries := database.New(db)


// THIS IS THE STATE WE ARE MANAGING IN OUR SERVER btw
cfg := &ApiConfig{
    db: dbQueries, // now all our func can use db methods e.g cfg.db.createUser()
}
```

# Server State Struct
```go
type ApiConfig struct {
	FileserverHits atomic.Int32 // allows us to increment int and read across go-routines
	// it is already 0 valid, so we can define our struct via
	// variableName := &ApiConfig
	db *database.Queries // place where our queries live
}
```
